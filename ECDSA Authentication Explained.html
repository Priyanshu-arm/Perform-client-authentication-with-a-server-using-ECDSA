Create a new file called keygen.cpp:

cpp
#include <stdio.h>
#include <stdint.h>
#include <string.h>
#include "ecdsa.h"
#include "secp256k1.h"
#include "bignum.h"

int main() {
    // 32-byte private key
    const char* private_key_hex = "1c3d5e7a9b2d4f6a1e3c5d7f9b2a4c6e8d0f123456789abcdef1234567890abcd";
    uint8_t private_key[32];
    hex_decode(private_key, 32, private_key_hex);

    // Calculate public key
    uint8_t public_key[65];
    ecdsa_get_public_key65(&secp256k1, private_key, public_key);

    printf("Private Key: %s\n", private_key_hex);
    printf("Public Key: ");
    for (int i = 0; i < 65; i++) {
        printf("%02x", public_key[i]);
    }
    printf("\n");

    return 0;
}

🧩 Step 3: Client Signs Serial ID

cpp
#include <iostream>
#include <string>
#include "sha2.h"
#include "ecdsa.h"
#include "secp256k1.h"
#include "bignum.h"

int main() {
    const std::string serial_id = "CLIENT12345678";

    // Hash serial ID using SHA256
    uint8_t hash[32];
    sha256_Raw((const uint8_t*)serial_id.c_str(), serial_id.length(), hash);

    // Private key
    const char* private_key_hex = "1c3d5e7a9b2d4f6a1e3c5d7f9b2a4c6e8d0f123456789abcdef1234567890abcd";
    uint8_t private_key[32];
    hex_decode(private_key, 32, private_key_hex);

    // Sign hash
    uint8_t signature[64];
    ecdsa_sign(&secp256k1, private_key, hash, signature, NULL);

    std::cout << "Signature: ";
    for (int i = 0; i < 64; ++i) {
        printf("%02x", signature[i]);
    }
    std::cout << std::endl;

    return 0;
}

🧩 Step 4: Server Verifies Signature

On server side, use:

cpp
bool valid = ecdsa_verify(&secp256k1, public_key, hash, signature);
if (valid) {
    std::cout << "✅ Signature is valid.\n";
} else {
    std::cout << "❌ Signature is NOT valid.\n";
}
